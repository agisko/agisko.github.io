Rozwiazanie:
 
--------------------------------------
#include <stdlib.h>
#include <stdio.h>
void main()
{
   int i, j, flaga, tablica[6];
   i = 0;
   j = 0;
   while(i < 6)
   { 
      srand(time(NULL));
      tablica[i] = rand() % 49 + 1;
      flaga = 0;
      for(j = 0; j < i; j++)
      {
         if(tablica[i] == tablica[j])
         { 
            flaga = 1;
            j = i;
         }
      }
      if(flaga == 0)
      {
         i++;
      }
   }   
   for(i = 0; i < 6; i++)
   {
      printf("%d\n", tablica[i]);
   }
}
